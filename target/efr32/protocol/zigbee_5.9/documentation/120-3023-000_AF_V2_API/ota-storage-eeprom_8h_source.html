51 // The Image Info Meta-data is DIFFERENT than previous versions of this plugin.
52 // The meta-data NOW contains the following data:
53 // Ember Magic Number (8 bytes)
54 // [different than the OTA file magic number]
55 // Version Tag (2 bytes)
56 // EBL Start offset within OTA file (4 bytes)
57 // Saved Download Offset data (100 bytes)
58 // This comes in 2 flavors:
59 // 1. For read-modify-write flash drivers this will be a simple 4-byte
60 // counter indicating the last downloaded offset. It will only be updated
61 // based on the plugin option...
62 // EMBER_AF_PLUGIN_OTA_STORAGE_SIMPLE_EEPROM_DOWNLOAD_OFFSET_SAVE_RATE
63 // Rest of the space is unused.
64 // 2. For flash drivers without read-modify-write support, we will use
65 // a byte-mask indicating the last full flash page of written data.
66 // The byte-mask will have negative logic (0xFF means flash page not
67 // downloaded) and requires one byte per page of the EEPROM space
68 // allocated for the OTA code.
69 // In other words if the client is given 200k of download space within
70 // the EEPROM, and a flash page is 4k, then we need 50 bytes for the
71 // byte-mask.

143 // The Offset within the OTA file (relative to offset 0) where the
144 // EBL data starts. We assume EBL data starts right after the 1st
145 // tag meta-data. However OTA headers are variable in size so we
146 // must keep track of how big the header is.
147 // This #define is actually the LOCATION where that offset is stored,
148 // not the actual offset.
149 #define EBL_START_OFFSET_INDEX (VERSION_NUMBER_INDEX + VERSION_NUMBER_SIZE)
150 #define EBL_START_OFFSET_SIZE 4
151 
152 // The last recorded offset we downloaded. This may not be the same as the
153 // value stored in RAM by the OTA cluster itself. This value will be
154 // the absolute offset of the file, regardless of the re-mapping
155 // this code does. It is an offset understood by the OTA storage interfaces
156 // relative to the start of the OTA file.
157 #define SAVED_DOWNLOAD_OFFSET_INDEX (EBL_START_OFFSET_INDEX + EBL_START_OFFSET_SIZE)
158 
159 // If we are using a Bytemask (Page erase required mode), then this is the maximum size.
160 // The bytemask limits the maximum size of the download image because it determines
161 // how many pages we can record as "fully downloaded". This allows the code to
162 // recover from a reboot in the middle of the download. A further complication is
163 // that this must be related to the word size of the EEPROM and the page size.
164 // In a perfect world we would size this based on the part but because the EEPROM
165 // info is likely contained within the bootloader we can't determine this except
166 // at runtime. This is sized for the worst case scenario as follows:
167 // 512k image size (3588) / 2048 page size (local storage bootloader)
168 // = 256 bytes * 2-byte word size = 512
169 // Most other parts use a 2048 or 4096 page size and are not 2-byte word sizes,
170 // so we are don't need this many bytes.
171 #define MAX_BYTEMASK_LENGTH 512
172 
173 #define OTA_HEADER_INDEX (SAVED_DOWNLOAD_OFFSET_INDEX + MAX_BYTEMASK_LENGTH)
