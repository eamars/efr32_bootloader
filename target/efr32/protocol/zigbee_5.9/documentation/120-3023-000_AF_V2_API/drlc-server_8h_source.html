2 // * drlc-server.h
3 // *
4 // * This is an implementation of a drlc server which is intended to
5 // * be used for testing only. Generally it is expected that the drlc
6 // * server will act as a passthrough for drlc events and will not maintain
7 // * a list of scheduled events of its own. However for testing purposes
8 // * it is highly useful to have the server store scheduled events so that
9 // * - minus a backend - the ESP can respond to get scheduled events
10 // * commands.
11 // *
12 // * There are a couple things about
13 // * this code that make it incomplete as an actual drlc server table
14 // *
15 // * 1. Scheduled events are not stored in persistent memory, which they
16 // * would have to be if this were intended for an ESP which for some
17 // * reason were storing events.
18 // *
19 // * Note: The EmberAfLoadControlEvent source field is used by the plugin as a
20 // * status field. If the first byte is set to 0xFF, then the event is
21 // * considered inactive, If it is set to 0x00, then the event is considered
22 // * active. Other bytes are used internally.
23 // *
24 // * Copyright 2007 by Ember Corporation. All rights reserved. *80*
25 // *******************************************************************
