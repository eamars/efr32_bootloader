68 /*
69 *********************************************************************************************************
70 * ASSERTS CONFIGURATION
71 *
72 * Note(s) : (1) RTOS_CFG_ASSERT_DBG_ARG_CHK_EXT_MASK is used to configure whether asserts used in
73 * argument checking are enabled or not. These asserts are used to check the validity of
74 * the arguments passed to any stack function. This can include checking if a pointer is
75 * non-NULL, if a given value is within a certain range, if a configuration is valid, etc.
76 *
77 * (a) This configuration define can be set as a bitmap of several module defines, to allow
78 * more flexibility. For example, it could be define as RTOS_CFG_MODULE_ALL or
79 * RTOS_CFG_MODULE_NONE to enable or disable it for every module. It could also be set
80 * as a combination of particular modules such as: (RTOS_CFG_MODULE_CPU |
81 * RTOS_CFG_MODULE_KERNEL), to keep the debug asserts only in the CPU and Kernel
82 * modules, but not in any other one.
83 *
84 * (b) It is recommended to disable this feature as much as possible in 'release' code, to
85 * improve performance and reduce the amount of code space required.
86 *
87 * (2) RTOS_CFG_RTOS_ASSERT_DBG_FAILED_END_CALL is used to configure what operations will
88 * happen in cases where a debug assert is failed. Debug asserts failing typically reflects
89 * an invalid argument passed by the programmer to a stack function or an operation that
90 * cannot be done in a particular context (such as an ISR). The program SHOULD NOT continue
91 * running after a debug assert failed, although it is possible to configure it to return
92 * (by defining it to: return ret_val (without parentheses or ';'), if the code is being
93 * executed in a test context. Typical operations could include breaking the CPU, looping
94 * indefinitely, calling a function or macro, etc. The default behavior is to loop
95 * indefinitely.
96 *
97 * (3) RTOS_CFG_RTOS_ASSERT_CRITICAL_FAILED_END_CALL is used to configure what operations will
98 * happen in cases where a critical assert is failed. Critical asserts failing reflects an
99 * unrecoverable situations in the system such as corruption or other cases that are
100 * unexpected. The program MUST NOT continue to execute in those cases. The program MUST
101 * NOT return from this call, since the software is in an unknown and/or invalid state.
102 * Typical operations could include outputting logs or traces, dumping memory, halting
103 * and/or restarting the system. The default behavior is to call the CPU_SW_EXCEPTION
104 * macro.
105 *********************************************************************************************************
106 */
107 
108 #define RTOS_CFG_ASSERT_DBG_ARG_CHK_EXT_MASK (RTOS_CFG_MODULE_ALL)
109 
110 #define RTOS_CFG_RTOS_ASSERT_DBG_FAILED_END_CALL(ret_val) while (1) {;}
111 
112 #define RTOS_CFG_RTOS_ASSERT_CRITICAL_FAILED_END_CALL(ret_val) CPU_SW_EXCEPTION(ret_val)

116 *********************************************************************************************************
117 * DFLT CFG CONFIGURATION
118 *
119 * Note(s) : (1) RTOS_CFG_EXTERNALIZE_OPTIONAL_CFG_EN determines whether the configurations can be
120 * provided via optional 'Configure' calls to the stacks or if these configurations are
121 * assumed extern by the stacks and must be provided by the application. If set to
122 * DEF_DISABLED, default configurations will be present in stacks and they can optionally
123 * be overridden with 'Configure' calls. This option is the simpler and should be the
124 * preferred way to use when starting development of an application. The default values
125 * used by the stacks are available for the application to copy, modify only parts of it
126 * and set as the new configuration. See each stack's user's manual for more information.
127 * If set to DEF_ENABLED, no default configuration will be present in the stacks and every
128 * configuration that could be set via 'Configure' calls are now assumed extern by the
129 * stacks and the application must define them somewhere. This option is useful to reduce
130 * the amount of memory and code space used.
131 *********************************************************************************************************
132 */
133 
134 #define RTOS_CFG_EXTERNALIZE_OPTIONAL_CFG_EN DEF_DISABLED

137 /*
138 *********************************************************************************************************
139 * LOGGING CONFIGURATION
140 *
141 * Note(s) : (1) RTOS_CFG_LOG_EN is used to enable or disable the logging module as a whole.
142 *
143 * (2) If enabled, RTOS_CFG_LOG_ALL MUST be defined. All logging channels will inherit this
144 * configuration if not overridden by a more specific configuration as shown below.
145 *
146 * (a) Values that can be used to configure logging channels are the following:
147 *
148 * (1) Lowest level to log:
149 * VRB Log error-, debug- and verbose-level messages.
150 * DBG Log error- and debug-level messages.
151 * ERR Log error-level messages only.
152 * OFF Do not log any level.
153 * DFLT Use the inherited setting.
154 *
155 * (2) Synchronicity of output:
156 * SYNC Output logs synchronously, during execution. May disrupt timing.
157 * ASYNC Save logs in buffer, always keeping the most recent ones.
158 * . Output will only be done when Log_Output() is called (see
159 * . rtos/common/include/logging.h for more details).
160 * DFLT Use the inherited setting.
161 *
162 * (3) Include function name in log entry:
163 * FUNC_EN Function name where entry was made will be included in entry.
164 * FUNC_DIS Function name where entry was made will NOT be included in entry.
165 * DFLT Use the inherited setting.
166 *
167 * (4) Include time-stamp in log entry:
168 * TS_EN Time-stamp of when entry was made will be included in entry.
169 * TS_DIS Time-stamp of when entry was made will NOT be included in entry.
170 * DFLT Use the inherited setting.
171 *
172 * (5) Log output function. Function must be of type 'int foo(int character)', as
173 * 'putchar', for example.
174 *
175 * (b) If RTOS_CFG_LOG_ALL is defined to: VRB, ASYNC, FUNC_DIS, TS_DIS, putchar and that
176 * no more configuration is defined, every logging channel will use this configuration,
177 * meaning that they will all output verbose (VRB) logs, in an asynchronous manner
178 * (ASYNC), without the function name (FUNC_DIS) or the timestamp (TS_DIS) using the
179 * 'putchar' function as output.
180 *
181 * (c) If another configuration is set for example for the kernel using the
182 * RTOS_CFG_LOG_KERNEL define (RTOS_CFG_LOG_ appended with the kernel's channel name,
183 * 'KERNEL') and that this define is set to ERR, ASYNC, DFLT, TS_EN, this would mean
184 * that the kernel channel (and only that channel) will output only error-level logs
185 * (ERR), asynchronously (ASYNC), use the inherited value for function name (DFLT) and
186 * override the timestamp configuration from RTOS_CFG_LOG_ALL by enabling it (TS_EN).
187 *********************************************************************************************************
188 */
189 
190 #define RTOS_CFG_LOG_EN DEF_DISABLED
191 
192 #if (RTOS_CFG_LOG_EN == DEF_ENABLED)
193 #include <stdio.h>
194 #define RTOS_CFG_LOG_ALL VRB, ASYNC, FUNC_DIS, TS_DIS, putchar
195 #endif
