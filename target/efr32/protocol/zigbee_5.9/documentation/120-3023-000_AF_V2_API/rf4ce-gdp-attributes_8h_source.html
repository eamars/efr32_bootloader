153 // Array attributes are in the range 0x90--0x9F for GDP and 0xC0--0xDF for the
154 // other profiles. This means that all array attributes have bit 7 set; bit 5
155 // clear; and bits 4 or 6 or both set. Note that this macro is just a tad not
156 // safe because it uses the parameter attributeId twice.
157 #define IS_ARRAY_ATTRIBUTE(attributeId) \
158  (READBITS((attributeId), BIT(7) | BIT(5)) == BIT(7) \
159  && READBITS((attributeId), BIT(4) | BIT(6)) != 0x00)
