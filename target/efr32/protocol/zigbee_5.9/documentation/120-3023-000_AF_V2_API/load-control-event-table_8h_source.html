2 // * load-control-event-table.h
3 // *
4 // * The Demand Response Load Control Event Table is responsible
5 // * for keeping track of all load control events scheduled
6 // * by the Energy Service Provider. This module provides
7 // * interfaces used to schedule and inform load shedding
8 // * devices of scheduled events.
9 // *
10 // * Any code that uses this event table is responsible for
11 // * providing four things:
12 // * 1. frequent calls to eventTableTick(), one per millisecond
13 // * will do. These calls are used to drive the
14 // * table's timing mechanism.
15 // * 2. A way to get the real time by implementing
16 // * getCurrentTime(uint32_t *currentTime);
17 // * 3. An implementation of eventAction which
18 // * will be called whenever event status changes
19 // *
20 // * The load control event table expects that currentTime, startTime
21 // * and randomization are provided in seconds. And that duration is
22 // * provided in minutes.
23 // *
24 // * The implementing code is responsible for over the
25 // * air communication based on event status changes
26 // * reported through the eventAction interface
27 // *
28 // * Copyright 2007 by Ember Corporation. All rights reserved. *80*
29 // *******************************************************************
