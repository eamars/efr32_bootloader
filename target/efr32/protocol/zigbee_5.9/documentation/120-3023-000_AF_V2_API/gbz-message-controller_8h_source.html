242 /*
243  * @brief This function initializes proper states for construction of GBZ messages.
244  *
245  * Depending on the value of the argument (gbzCommand), the creator will
246  * behave differently. If a NULL value is passed, the creator assumes the
247  * user wants the API to allocate memory to store the appended responses.
248  * Otherwise, the creator will use the provided buffer as the destination for
249  * storing responses.
250  *
251  * Below is a general flow for the creation of ZCL messages into a GBZ
252  * message.
253  *
254  * 1. emberAfPluginGbzMessageControllerCreatorInit() - create
255  * 2. emberAfPluginGbzMessageControllerAppendCommand() - append
256  * 3. emberAfPluginGbzMessageControllerCreatorAssemble() - assemble result.
257  * 4. emberAfPluginGbzMessageControllerCreatorCleanup() - memory clean up
258  *
259  * @param state A pre-allocated struct that will be updated to hold
260  * bookkeping information for creating GBZ messages
261  * @param type GBZ payload type: command, response, or alert
262  * @param alertCode When type is alert this field contains the alert code
263  * @param timestamp When type is alert this field contains the UTC when the alert occurred
264  * @param gbzCommand NULL - if the user wants API to allocate memory to store
265  * respones
266  * Otherwsie - pointer to destination buffer for gbz messages
267  * @param gbzCommandLength length of the gbz messages. this argument is
268  * ignored if gbzCommand is NULL
269  *
270  */
271 uint16_t emberAfPluginGbzMessageControllerCreatorInit(EmberAfGbzMessageCreatorState * state,
272  EmberAfGbzMessageType type,
273  uint16_t alertCode,
274  uint32_t timestamp,
275  uint16_t messageCode,
276  uint8_t * gbzCommand,
277  uint16_t gbzCommandLength);
