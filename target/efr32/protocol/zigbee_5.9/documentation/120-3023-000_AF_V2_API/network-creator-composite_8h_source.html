25 // Here are the threshold values for the channel composite struct. These enable
26 // the network creator to use discrete logic to find the best channel on which
27 // to form a network. If a channel has a composite where one of the values is
28 // higher than the corresponding threshold, then the network creator will not
29 // form a network on this channel.
30 #ifndef EMBER_AF_PLUGIN_NETWORK_CREATOR_CHANNEL_BEACONS_THRESHOLD
31  #define EMBER_AF_PLUGIN_NETWORK_CREATOR_CHANNEL_BEACONS_THRESHOLD 5
32 #endif
33 
34 #define emAfPluginNetworkCreatorChannelCompositeIsAboveThreshold(composite) \
35  (((composite).beaconsHeard \
36  > EMBER_AF_PLUGIN_NETWORK_CREATOR_CHANNEL_BEACONS_THRESHOLD))
