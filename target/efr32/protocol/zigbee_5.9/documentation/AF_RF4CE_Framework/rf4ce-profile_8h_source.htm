708 // There are a maximum of 128 bytes at the PHY layer, but there is a one-byte
709 // length and a two-byte CRC, leaving 125 for the PHY payload. The minimum MAC
710 // header consists of a two-byte frame control, one-byte sequence number, two-
711 // byte destination pan id, and at least two bytes each for the source and
712 // destination addresses. That leaves a maximum of 116 bytes for the MAC
713 // payload, although inter-pan messages or long source or destination addresses
714 // will reduce that. The minimum NWK header consists of a one-byte frame
715 // control, four-byte frame counter, and one-byte profile id. That leaves a
716 // maximum of 110 bytes for the NWK payload, although encrypted or vendor-
717 // specific messages will reduce that. Use the ::emberAfRf4ceGetMaxPayload()
718 // API to retrieve the maximum allowed payload for a certain pairing and certain
719 // TX options.
720 #define EMBER_AF_RF4CE_MAXIMUM_RF4CE_PAYLOAD_LENGTH 110
