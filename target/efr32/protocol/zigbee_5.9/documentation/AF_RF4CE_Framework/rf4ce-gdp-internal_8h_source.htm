374 // Setting the public state clears the internal state (implicitly set to NONE).
375 #define setPublicState(state, init) \
376  do { \
377  emAfGdpState = (state); \
378  if (!init) { \
379  emberAfRf4ceRxEnable(EMBER_AF_RF4CE_PROFILE_GENERIC_DEVICE, false); \
380  } \
381  } while(0)
382 
383 // For any state other than NONE we keep the receiver ON.
384 // The "STACK_STATUS_NETWORK_UP_PENDING" is a special state that is set in the
385 // init function at the recipient. Since we can not control the order of the
386 // plugin init functions calls, we might end up calling the RxEnable before the
387 // profile plugin has been initialized. For this reason, we don't call the
388 // RxEnable() in this state.
389 #define setInternalState(state) \
390  do { \
391  emAfGdpState = ((emAfGdpState & PUBLIC_STATE_MASK) | (state)); \
392  if ((state) != INTERNAL_STATE_RECIPIENT_GDP_STACK_STATUS_NETWORK_UP_PENDING) { \
393  emberAfRf4ceRxEnable(EMBER_AF_RF4CE_PROFILE_GENERIC_DEVICE, \
394  ((state) != INTERNAL_STATE_NONE)); \
395  } \
396  } while(0)
